Mi az a konstans?

Egy olyan érték, amit a program elején beállítasz, és SOHA nem változtatod meg.

def calculate_class_average(student_grades):
    """Kiszámolja egy osztály átlagát.

    Visszaadja: az átlagot (szám)

def find_best_student(student_grades):
    """Megkeresi a legjobb jegyű diákot.

    Kapja: szótár, pl. {"Anna": 4, "Béla": 3}
    Visszaadja: a legjobb diák nevét (szöveg)

        0.0
    """
    discount_amount: float = price * (discount_percent / 100)   # kedvezmény összege
    final_price: float = price - discount_amount                # kedvezményes ár

    if final_price < 0:        # ha a kedvezmény túl nagy volt
        return 0.0             # ne legyen negatív ár

    return final_price         # visszaadjuk a kedvezményes árat

def read_and_convert(file_name: str, index: int) -> str:
    """Egy fájlt olvas és a tartalmát számmá alakítja.

fake_files: dict[str, list[str]] = {

def set_product_price(name: str, price: int) -> str:
    """Beállítja egy termék árát.

    Az ár nem lehet negatív és nem lehet nulla,
    mert ingyen nem adunk el semmit.
    """
    if not isinstance(price, (int, float)):    # ellenőrizzük, hogy szám-e

        raise ValueError("Az ár nem lehet negatív!")

    return f"'{name}' ára beállítva: {price} Ft"
